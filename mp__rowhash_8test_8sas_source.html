19/* test 1 - versioned seed only (no input variables) */
20data _null_;
21 length actual expected $32;
22 actual='';
23 %mp_rowhash(md5_col=actual)
24 expected=put(md5('DC HASH v2'),$hex32.);
25 call symputx('t1_actual',actual);
26 call symputx('t1_expected',expected);
27run;
28%mp_assertscope(COMPARE,ignorelist=T1_ACTUAL T1_EXPECTED)
29%mp_assert(
30 iftrue=("&t1_actual"="&t1_expected"),
31 desc=Versioned seed is returned when no variables are supplied,
32 outds=work.test_results
33)

35/* test 2 - single character variable reference */
36data _null_;
37 length c1 $1 actual expected $32;
38 length state digest $16 pair $32;
39 c1='A';
40 actual='';
41 %mp_rowhash(md5_col=actual, cvars=c1)
42 state=md5('DC HASH v2');
43 digest=md5(trimn(c1));
44 pair=state||digest;
45 expected=put(md5(pair),$hex32.);

55/* test 3 - single numeric variable reference */
56data _null_;
57 length actual expected $32;
58 length state digest $16 numtext $64 pair $32;
59 n1=42;
60 actual='';
61 %mp_rowhash(md5_col=actual, nvars=n1)
62 state=md5('DC HASH v2');
63 /* multiply-by-one matches the internal normalisation */
64 normal=n1*1;
65 numtext=put(normal,binary64.);
66 digest=md5(trim(numtext));
67 pair=state||digest;
68 expected=put(md5(pair),$hex32.);
69 call symputx('t3_actual',actual);
70 call symputx('t3_expected',expected);
71run;
