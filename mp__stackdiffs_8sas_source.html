1/**
2 @file
3 @brief Prepares an audit table for stacking (re-applying) the changes. WORK
4 IN PROGRESS!!
5 @details WORK IN PROGRESS!!
6 When a Base Table is refreshed, it can be helpful to have any
7 subsequent changes re-applied. This macro is used to convert stored changes
8 (tall format) into staged changes (wide format), with base table values
9 incorporated, ready for loading.
10
11 The audit table is assumed to be structured as per the mp_storediffs.sas
12 macro (DIFFTABLE entry).
13
14 The macro supports both SCD2 (TXTEMPORAL) and UPDATE loadtypes. If the
15 base table contains a PROCESSED_DTTM column (or similar), this can be
16 ignored by declaring it in the `processed_dttm_var` parameter.
17
18 If the Base Table has changed in structure, the following rules apply:
19
20 @li New Columns - Ignored for deletes, set to missing for inserts. For
21 updates, base table values are used.
22 @li Deleted Columns - These are simply removed from the staging table.
23 @li Change in Primary Key - this is not (well, cannot be) supported!!
24
25 Rules for stacking changes are as follows:
26
27 <table>
28 <tr><th>Transaction Type</th><th>Behaviour</th></tr>
29 <tr>
30 <td>Deletes</td>
31 <td>
32 For previously deleted rows, the PK is added to the staging table and
33 the `&deletecol` variable is set to "Yes".<br>
34 If the row no
35 longer exists in the base table, the row is added to the errds table
36 instead.
37 </td>
38 </tr>
39 <tr>
40 <td>Inserts</td>
41 <td>
42 Previously newly added rows are added to the staging table (with
43 `&deletecol` set to "No") UNLESS they are present in the Base table.
44 <br>In this case they are added to the errds table instead.
45 </td>
46 </tr>
47 <tr>
48 <td>Updates</td>
49 <td>
50 Previously modified rows are merged with base table values such that
51 only the individual cells that were _previously_ changed are re-applied.
52 Where the row contains cells that were not marked as having changed in
53 the prior transaction, the 'blanks' are filled with base table values in
54 the staging table.<br>
55 If the row no longer exists on the base table, then the row is added to
56 the errds table instead.
57 </td>
58 </tr>
59 </table>
60
61 To illustrate the above with a diagram:
62
63 @dot
64 digraph {
65 rankdir="TB"
66
67 start[label="Transaction Type?" shape=Mdiamond]
68 del[label="Base Row Exists?" shape=rectangle]
69 add [label="Base Row Exists?" shape=rectangle]
70 mod [label="Base Row Exists?" shape=rectangle]
71 stg [label="Stage Table" shape=box3d]
72 err [label="ErrDS Table" shape=box3d]
73 start -> del [label="Delete"]
74 start -> add [label="Insert"]
75 start -> mod [label="Update"]
76
77 del -> stg [label="Yes"]
78 del -> err [label="No" color="Red" fontcolor="Red"]
79 add -> stg [label="Yes"]
80 add -> err [label="No" color="Red" fontcolor="Red"]
81 mod -> err [label="Yes" color="Red" fontcolor="Red"]
82 mod -> stg [label="No"]
83
84 }
85 @enddot
86
87 For examples of usage, check out the mp_stackdiffs.test.sas program.
88

90 @param [in] baselibds Base Table against which the changes will be applied,
91 in libref.dataset format.
92 @param [in] auditlibds Dataset with previously applied transactions, to be
93 re-applied. Use libref.dataset format.
94 DDL as follows: %mp_coretable(DIFFTABLE)
95 @param [in] key Space seperated list of key variables
96 @param [in] delete_col= (_____DELETE__THIS__RECORD_____) Variable name used
97 to mark the row as being deleted
98 @param [in] mdebug= set to 1 to enable DEBUG messages and preserve outputs
99 @param [in] processed_dttm_var= (0) If a variable is being used to mark
100 the processed datetime, put the name of the variable here. It will NOT
101 be included in the staged dataset (the load process is expected to
102 provide this)
103 @param [out] outds= (work.mp_stackdiffs) Output table containing staged data
104 to be re-applied.

122%macro mp_stackdiffs(baselibds
123 ,auditlibds
124 ,key
125 ,delete_col=_____DELETE__THIS__RECORD_____
126 ,mdebug=0
127 ,processed_dttm_var=0
128 ,outds=work.mp_stackdiffs
129)/*/STORE SOURCE*/;
130%local dbg;
131%if &mdebug=1 %then %do;
132 %put &sysmacroname entry vars:;
133 %put _local_;
134%end;
135%else %let dbg=*;
136
137/* input parameter validations */
138%mp_abort(iftrue= (%mf_islibds(&baselibds) ne 1)
139 ,mac=&sysmacroname
140 ,msg=%str(Invalid baselibds: &baselibds)
141)
142%mp_abort(iftrue= (%mf_islibds(&auditlibds) ne 1)
143 ,mac=&sysmacroname
144 ,msg=%str(Invalid auditlibds: &auditlibds)
145)
146%mp_abort(iftrue= (%length(&key)=0)
147 ,mac=&sysmacroname
148 ,msg=%str(Missing key variables!)
149)
150
151/* set up unique and temporary vars */
152%local ds1 ds2 ds3 ds4 hashkey inds_auto inds_keep dslist;
153%let ds1=%upcase(work.%mf_getuniquename(prefix=mpsd_ds1));
154%let ds2=%upcase(work.%mf_getuniquename(prefix=mpsd_ds2));
155%let ds3=%upcase(work.%mf_getuniquename(prefix=mpsd_ds3));
156%let ds4=%upcase(work.%mf_getuniquename(prefix=mpsd_ds4));
157%let hashkey=%upcase(%mf_getuniquename(prefix=mpsd_hashkey));
158%let inds_auto=%upcase(%mf_getuniquename(prefix=mpsd_inds_auto));
159%let inds_keep=%upcase(%mf_getuniquename(prefix=mpsd_inds_keep));
160
161%let key=%upcase(&key);
162
163/* hash the key and append all the tables (marking the source) */
164data &ds1;
165 set &dslist indsname=&inds_auto;
166 &hashkey=put(md5(catx('|',%mf_getquotedstr(&key,quote=N))),$hex32.);
167 &inds_keep=&inds_auto;
168proc sort;
169 by &inds_keep &hashkey;
170run;
171
172/* transpose numeric & char vars */
173proc transpose data=&ds1
174 out=&ds2(rename=(&hashkey=key_hash _name_=tgtvar_nm col1=newval_num));
175 by &inds_keep &hashkey;
176 var _numeric_;
177run;
178proc transpose data=&ds1
179 out=&ds3(
180 rename=(&hashkey=key_hash _name_=tgtvar_nm col1=newval_char)
181 where=(tgtvar_nm not in ("&hashkey","&inds_keep"))
182 );
183 by &inds_keep &hashkey;
184 var _character_;
185run;
186data &ds4;
187 length &inds_keep $41 tgtvar_nm $32;
188 set &ds2 &ds3 indsname=&inds_auto;
189
190 tgtvar_nm=upcase(tgtvar_nm);
191 if tgtvar_nm in (%upcase(%mf_getvarlist(&libds,dlm=%str(,),quote=DOUBLE)));
192
193 if &inds_auto="&ds2" then tgtvar_type='N';
194 else if &inds_auto="&ds3" then tgtvar_type='C';
195 else do;
196 putlog "%str(ERR)OR: unidentified vartype input!" &inds_auto;
197 call symputx('syscc',98);
198 end;
199
200 if &inds_keep="&appds" then move_type='A';
201 else if &inds_keep="&delds" then move_type='D';
202 else if &inds_keep="&modds" then move_type='M';
203 else if &inds_keep="&origds" then move_type='O';
204 else do;
205 putlog "%str(ERR)OR: unidentified movetype input!" &inds_keep;
206 call symputx('syscc',99);
207 end;
208 tgtvar_nm=upcase(tgtvar_nm);
209 if tgtvar_nm in (%mf_getquotedstr(&key)) then is_pk=1;
210 else is_pk=0;
211 drop &inds_keep;
212run;
213
214%if "&loadref"="0" %then %let loadref=%sysfunc(uuidgen());
215%if &processed_dttm=0 %then %let processed_dttm=%sysfunc(datetime());
216%let libds=%upcase(&libds);
217
218/* join orig vals for modified & deleted */
219proc sql;
220create table &outds as
221 select "&loadref" as load_ref length=36
222 ,&processed_dttm as processed_dttm format=E8601DT26.6
223 ,"%scan(&libds,1,.)" as libref length=8
224 ,"%scan(&libds,2,.)" as dsn length=32
225 ,b.key_hash length=32
226 ,b.move_type length=1
227 ,b.tgtvar_nm length=32
228 ,b.is_pk
229 ,case when b.move_type ne 'M' then -1
230 when a.newval_num=b.newval_num and a.newval_char=b.newval_char then 0
231 else 1
232 end as is_diff
233 ,b.tgtvar_type length=1
234 ,case when b.move_type='D' then b.newval_num
235 else a.newval_num
236 end as oldval_num format=best32.
237 ,case when b.move_type='D' then .
238 else b.newval_num
239 end as newval_num format=best32.
240 ,case when b.move_type='D' then b.newval_char
241 else a.newval_char
242 end as oldval_char length=32765
243 ,case when b.move_type='D' then ''
244 else b.newval_char
245 end as newval_char length=32765
246 from &ds4(where=(move_type='O')) as a
247 right join &ds4(where=(move_type ne 'O')) as b
248 on a.tgtvar_nm=b.tgtvar_nm
249 and a.key_hash=b.key_hash
250 order by move_type, key_hash,is_pk desc, tgtvar_nm;
251
252%if &mdebug=0 %then %do;
253 proc sql;
254 drop table &ds1, &ds2, &ds3, &ds4;
255%end;
256
257%mend mp_stackdiffs;
258/** @endcond */
