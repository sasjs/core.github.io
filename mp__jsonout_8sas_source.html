35  @param [in] action Valid values:
36  @li OPEN - opens the JSON
37  @li OBJ - sends a table with each row as an object
38  @li ARR - sends a table with each row in an array
39  @li CLOSE - closes the JSON
40  @param [in] ds The dataset to send. Must be a work table.
41  @param [out] jref= (_webout) The fileref to which to send the JSON
42  @param [out] dslabel= The name to give the table in the exported JSON
43  @param [in] fmt= (Y) Whether to keep (Y) or strip (N) formats from the table
44  @param [in] engine= (DATASTEP) Which engine to use to send the JSON. Options:
45  @li PROCJSON (default)
46  @li DATASTEP (more reliable when data has non standard characters)

49  @param [in] showmeta= (NO) Set to YES to output metadata alongside each table,

84  out=_data_(keep=name type length format formatl formatd varnum label);
85  run;
86  %let colinfo=%scan(&syslast,2,.);
87  proc sort data=&colinfo;
88  by varnum;
89  run;

91  data _null_;
92  if _n_=1 then call symputx('numcols',nobs,'l');
93  set &colinfo end=last nobs=nobs;
94  name=upcase(name);
95  /* fix formats */
96  if type=2 or type=6 then do;
97  typelong='char';
98  length fmt $49.;
99  if format='' then fmt=cats('$',length,'.');
100  else if formatl=0 then fmt=cats(format,'.');
101  else fmt=cats(format,formatl,'.');
102  newlen=max(formatl,length);
103  end;
104  else do;
105  typelong='num';
106  if format='' then fmt='best.';
107  else if formatl=0 then fmt=cats(format,'.');
108  else if formatd=0 then fmt=cats(format,formatl,'.');
109  else fmt=cats(format,formatl,'.',formatd);
110  /* needs to be wide, for datetimes etc */
111  newlen=max(length,formatl,24);
112  end;
113  /* 32 char unique name */
114  newname='sasjs'!!substr(cats(put(md5(name),$hex32.)),1,27);
115 
116  call symputx(cats('name',_n_),name,'l');
117  call symputx(cats('newname',_n_),newname,'l');
118  call symputx(cats('len',_n_),newlen,'l');
119  call symputx(cats('length',_n_),length,'l');
120  call symputx(cats('fmt',_n_),fmt,'l');
121  call symputx(cats('type',_n_),type,'l');
122  call symputx(cats('typelong',_n_),typelong,'l');
123  call symputx(cats('label',_n_),coalescec(label,name),'l');
124  run;

164  &&name&i=put(&&newname&i,&&fmt&i);
