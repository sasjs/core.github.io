59%macro mp_getformats(lib=0
60 ,ds=0
61 ,fmtlist=0
62 ,outsummary=work.mp_getformats_summary
63 ,outdetail=0
64);
65
66%local i fmt allfmts tempds fmtcnt;
67
68%if "&fmtlist" ne "0" %then %do i=1 %to %sysfunc(countw(&fmtlist,,%str( )));
69 /* ensure format list contains format _name_ only */
70 %let fmt=%scan(&fmtlist,&i,%str( ));
71 %let fmt=%mf_getfmtname(&fmt);
72 %let allfmts=&allfmts &fmt;
73%end;
74
75%if &ds=0 and &lib ne 0 %then %do;
76 /* grab formats from library */
77 /* to do */
78%end;
79%else %if &ds ne 0 and &lib ne 0 %then %do;
80 /* grab formats from dataset */
81 %let allfmts=%mf_getfmtlist(&lib..&ds) &allfmts;
82%end;
83
84/* ensure list is unique */
85%let allfmts=%mf_dedup(%upcase(&allfmts));
86
87/* create summary table */
88%if %index(&outsummary,.)=0 %then %let outsummary=WORK.&outsummary;
89proc sql;
90create table &outsummary as
91 select * from dictionary.formats
92 where fmtname in (%mf_getquotedstr(&allfmts,quote=D))
93 and fmttype='F';
94
95%if "&outdetail" ne "0" %then %do;
96 /* ensure base table always exists */
97 proc sql;
98 create table &outdetail(
99 FMTNAME char(32) label='Format name'
100 ,START char(16) label='Starting value for format'
101 ,END char(16) label='Ending value for format'
102 ,LABEL char(256) label='Format value label'
103 ,MIN num length=3 label='Minimum length'
104 ,MAX num length=3 label='Maximum length'
105 ,DEFAULT num length=3 label='Default length'
106 ,LENGTH num length=3 label='Format length'
107 ,FUZZ num label='Fuzz value'
108 ,PREFIX char(2) label='Prefix characters'
109 ,MULT num label='Multiplier'
110 ,FILL char(1) label='Fill character'
111 ,NOEDIT num length=3 label='Is picture string noedit?'
112 ,TYPE char(1) label='Type of format'
113 ,SEXCL char(1) label='Start exclusion'
114 ,EEXCL char(1) label='End exclusion'
115 ,HLO char(13) label='Additional information'
116 ,DECSEP char(1) label='Decimal separator'
117 ,DIG3SEP char(1) label='Three-digit separator'
118 ,DATATYPE char(8) label='Date/time/datetime?'
119 ,LANGUAGE char(8) label='Language for date strings'
120 );
121 /* grab the location of each format */
122 %let fmtcnt=0;
123 data _null_;
124 set &outsummary;
125 if not missing(libname);
126 x+1;
127 call symputx(cats('fmtloc',x),cats(libname,'.',memname),'l');
128 call symputx(cats('fmtname',x),fmtname,'l');
129 call symputx('fmtcnt',x,'l');
130 run;
131 /* export each format and append to the output table */
132 %let tempds=%mf_getuniquename(prefix=mp_getformats);
133 %do i=1 %to &fmtcnt;
134 proc format library=&&fmtloc&i CNTLOUT=&tempds;
135 select &&fmtname&i;
136 run;
137 proc append base=&outdetail data=&tempds;
138 run;
139 %end;
140%end;
141
142%mend mp_getformats;
