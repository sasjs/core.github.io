66function mcf_getfmttype(fmtnm $) $8;
67 if substr(fmtnm,1,1)='$' then return('CHAR');
68 else do;
69 /* extract NAME */
70 length fmt $32;
71 fmt=scan(fmtnm,1,'.');
72 do while (
73 substr(fmt,length(fmt),1) in ('1','2','3','4','5','6','7','8','9','0')
74 );
75 if length(fmt)=1 then fmt='W';
76 else fmt=substr(fmt,1,length(fmt)-1);
77 end;
78
79 /* apply lookups */
80 if cats(fmt) in ('DATETIME','B8601DN','B8601DN','B8601DT','B8601DT'
81 ,'B8601DZ','B8601DZ','DATEAMPM','DTDATE','DTMONYY','DTWKDATX','DTYEAR'
82 ,'DTYYQC','E8601DN','E8601DN','E8601DT','E8601DT','E8601DZ','E8601DZ')
83 then return('DATETIME');
84 else if fmt in ('DATE','YYMMDD','B8601DA','B8601DA','DAY','DDMMYY'
85 ,'DDMMYYB','DDMMYYC','DDMMYYD','DDMMYYN','DDMMYYP','DDMMYYS','DDMMYYx'
86 ,'DOWNAME','E8601DA','E8601DA','JULDAY','JULIAN','MMDDYY','MMDDYYB'
87 ,'MMDDYYC','MMDDYYD','MMDDYYN','MMDDYYP','MMDDYYS','MMDDYYx','MMYY'
88 ,'MMYYC','MMYYD','MMYYN','MMYYP','MMYYS','MMYYx','MONNAME','MONTH'
89 ,'MONYY','PDJULG','PDJULI','QTR','QTRR','WEEKDATE','WEEKDATX','WEEKDAY'
90 ,'WEEKU','WEEKV','WEEKW','WORDDATE','WORDDATX','YEAR','YYMM','YYMMC'
91 ,'YYMMD','YYMMDDB','YYMMDDC','YYMMDDD','YYMMDDN','YYMMDDP','YYMMDDS'
92 ,'YYMMDDx','YYMMN','YYMMP','YYMMS','YYMMx','YYMON','YYQ','YYQC','YYQD'
93 ,'YYQN','YYQP','YYQR','YYQRC','YYQRD','YYQRN','YYQRP','YYQRS','YYQRx'
94 ,'YYQS','YYQx','YYQZ') then return('DATE');
95 else if fmt in ('TIME','B8601LZ','B8601LZ','B8601TM','B8601TM','B8601TZ'
96 ,'B8601TZ','E8601LZ','E8601LZ','E8601TM','E8601TM','E8601TZ','E8601TZ'
97 ,'HHMM','HOUR','MMSS','TIMEAMPM','TOD') then return('TIME');
98 else return('NUM');
99 end;
