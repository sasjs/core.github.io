2 @file
3 @brief Send data to/from @sasjs/server
4 @details This macro should be added to the start of each web service,
5 **immediately** followed by a call to:
6
7 %ms_webout(FETCH)
8
9 This will read all the input data and create same-named SAS datasets in the
10 WORK library. You can then insert your code, and send data back using the
11 following syntax:
12
13 data some datasets; * make some data ;
14 retain some columns;
15 run;
16
17 %ms_webout(OPEN)
18 %ms_webout(ARR,some) * Array format, fast, suitable for large tables ;
19 %ms_webout(OBJ,datasets) * Object format, easier to work with ;
20 %ms_webout(CLOSE)
21
22
23 @param action Either FETCH, OPEN, ARR, OBJ or CLOSE
24 @param ds The dataset to send back to the frontend
25 @param dslabel= value to use instead of the real name for sending to JSON
26 @param fmt=(Y) Set to N to send back unformatted values
27 @param fref=(_webout) The fileref to which to write the JSON

42%macro ms_webout(action,ds,dslabel=,fref=_webout,fmt=Y);
43%global _webin_file_count _webin_fileref1 _webin_name1 _program _debug
44 sasjs_tables;
45
46%local i tempds;
47%let action=%upcase(&action);
48
49%if &action=FETCH %then %do;
50 %if %str(&_debug) ge 131 %then %do;
51 options mprint notes mprintnest;
52 %end;
53 %let _webin_file_count=%eval(&_webin_file_count+0);
54 /* now read in the data */
55 %do i=1 %to &_webin_file_count;
56 %if &_webin_file_count=1 %then %do;
57 %let _webin_fileref1=&_webin_fileref;
58 %let _webin_name1=&_webin_name;
59 %end;
60 data _null_;
61 infile &&_webin_fileref&i termstr=crlf;
62 input;
63 call symputx('input_statement',_infile_);
64 putlog "&&_webin_name&i input statement: " _infile_;
65 stop;
66 data &&_webin_name&i;
67 infile &&_webin_fileref&i firstobs=2 dsd termstr=crlf encoding='utf-8';
68 input &input_statement;
69 %if %str(&_debug) ge 131 %then %do;
70 if _n_<20 then putlog _infile_;
71 %end;
72 run;
73 %let sasjs_tables=&sasjs_tables &&_webin_name&i;
74 %end;
75%end;
76
77%else %if &action=OPEN %then %do;
78 /* fix encoding */
79 OPTIONS NOBOMFILE;
80
81 /* setup json */
82 data _null_;file &fref encoding='utf-8' termstr=lf;
83 %if %str(&_debug) ge 131 %then %do;
84 put '>>weboutBEGIN<<';
85 %end;
86 put '{"SYSDATE" : "' "&SYSDATE" '"';
87 put ',"SYSTIME" : "' "&SYSTIME" '"';
88 run;
89
90%end;
91
92%else %if &action=ARR or &action=OBJ %then %do;
93 %mp_jsonout(&action,&ds,dslabel=&dslabel,fmt=&fmt,jref=&fref
94 ,engine=DATASTEP,dbg=%str(&_debug)
95 )
96%end;
97%else %if &action=CLOSE %then %do;
98 %if %str(&_debug) ge 131 %then %do;
99 /* if debug mode, send back first 10 records of each work table also */
100 options obs=10;
101 data;run;%let tempds=%scan(&syslast,2,.);
102 ods output Members=&tempds;
103 proc datasets library=WORK memtype=data;
104 %local wtcnt;%let wtcnt=0;
105 data _null_;
106 set &tempds;
107 if not (upcase(name) =:"DATA"); /* ignore temp datasets */
108 i+1;
109 call symputx('wt'!!left(i),name,'l');
110 call symputx('wtcnt',i,'l');
111 data _null_; file &fref mod encoding='utf-8' termstr=lf;
112 put ",""WORK"":{";
113 %do i=1 %to &wtcnt;
114 %let wt=&&wt&i;
115 proc contents noprint data=&wt
116 out=_data_ (keep=name type length format:);
117 run;%let tempds=%scan(&syslast,2,.);
118 data _null_; file &fref mod encoding='utf-8' termstr=lf;
119 dsid=open("WORK.&wt",'is');
120 nlobs=attrn(dsid,'NLOBS');
121 nvars=attrn(dsid,'NVARS');
122 rc=close(dsid);
123 if &i>1 then put ','@;
124 put " ""&wt"" : {";
125 put '"nlobs":' nlobs;
126 put ',"nvars":' nvars;
127 %mp_jsonout(OBJ,&tempds,jref=&fref,dslabel=colattrs,engine=DATASTEP)
128 %mp_jsonout(OBJ,&wt,jref=&fref,dslabel=first10rows,engine=DATASTEP)
129 data _null_; file &fref mod encoding='utf-8' termstr=lf;
130 put "}";
131 %end;
132 data _null_; file &fref mod encoding='utf-8' termstr=lf termstr=lf;
133 put "}";
134 run;
135 %end;
136 /* close off json */
137 data _null_;file &fref mod encoding='utf-8' termstr=lf;
138 _PROGRAM=quote(trim(resolve(symget('_PROGRAM'))));
139 put ",""SYSUSERID"" : ""&sysuserid"" ";
140 put ",""MF_GETUSER"" : ""%mf_getuser()"" ";
141 put ",""_DEBUG"" : ""&_debug"" ";
142 put ',"_PROGRAM" : ' _PROGRAM ;
143 put ",""SYSCC"" : ""&syscc"" ";
144 put ",""SYSERRORTEXT"" : ""&syserrortext"" ";
145 SYSHOSTINFOLONG=quote(trim(symget('SYSHOSTINFOLONG')));
146 put ',"SYSHOSTINFOLONG" : ' SYSHOSTINFOLONG;
147 put ",""SYSHOSTNAME"" : ""&syshostname"" ";
148 put ",""SYSPROCESSID"" : ""&SYSPROCESSID"" ";
149 put ",""SYSPROCESSMODE"" : ""&SYSPROCESSMODE"" ";
150 length SYSPROCESSNAME $512;
151 SYSPROCESSNAME=quote(urlencode(cats(SYSPROCESSNAME)));
152 put ",""SYSPROCESSNAME"" : " SYSPROCESSNAME;
153 put ",""SYSJOBID"" : ""&sysjobid"" ";
154 put ",""SYSSCPL"" : ""&sysscpl"" ";
155 put ",""SYSSITE"" : ""&syssite"" ";
156 put ",""SYSTCPIPHOSTNAME"" : ""&SYSTCPIPHOSTNAME"" ";
157 sysvlong=quote(trim(symget('sysvlong')));
158 put ',"SYSVLONG" : ' sysvlong;
159 put ",""SYSWARNINGTEXT"" : ""&syswarningtext"" ";
160 put ',"END_DTTM" : "' "%sysfunc(datetime(),datetime20.3)" '" ';
161 length autoexec $512;
162 autoexec=quote(urlencode(trim(getoption('autoexec'))));
163 put ',"AUTOEXEC" : ' autoexec;
164 memsize="%sysfunc(INPUTN(%sysfunc(getoption(memsize)), best.),sizekmg.)";
165 memsize=quote(cats(memsize));
166 put ',"MEMSIZE" : ' memsize;
167 put "}" @;
168 %if %str(&_debug) ge 131 %then %do;
169 put '>>weboutEND<<';
170 %end;
171 run;
172%end;

174%mend ms_webout;
