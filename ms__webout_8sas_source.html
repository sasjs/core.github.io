1/**
2 @file
3 @brief Send data to/from sasjs/server
4 @details This macro should be added to the start of each web service,
5 **immediately** followed by a call to:
6
7 %ms_webout(FETCH)
8
9 This will read all the input data and create same-named SAS datasets in the
10 WORK library. You can then insert your code, and send data back using the
11 following syntax:
12
13 data some datasets; * make some data ;
14 retain some columns;
15 run;
16
17 %ms_webout(OPEN)
18 %ms_webout(ARR,some) * Array format, fast, suitable for large tables ;
19 %ms_webout(OBJ,datasets) * Object format, easier to work with ;
20 %ms_webout(CLOSE)
21
22
23 @param [in] action Either FETCH, OPEN, ARR, OBJ or CLOSE
24 @param [in] ds The dataset to send back to the frontend
25 @param [out] dslabel= value to use instead of table name for sending to JSON
26 @param [in] fmt= (Y) Set to N to send back unformatted values
27 @param [out] fref= (_webout) The fileref to which to write the JSON
28 @param [in] missing= (NULL) Special numeric missing values can be sent as NULL
29 (eg `null`) or as STRING values (eg `".a"` or `".b"`)
30 @param [in] showmeta= (NO) Set to YES to output metadata alongside each table,
31 such as the column formats and types. The metadata is contained inside an
32 object with the same name as the table but prefixed with a dollar sign - ie,
33 `,"$tablename":{"formats":{"col1":"$CHAR1"},"types":{"COL1":"C"}}`
34
35 <h4> SAS Macros </h4>
36 @li mf_getuser.sas
37 @li mp_jsonout.sas
38 @li mfs_httpheader.sas
39
40 <h4> Related Macros </h4>
41 @li mv_webout.sas
42 @li mm_webout.sas
43
44 @version 9.3
45 @author Allan Bowe
46
47**/

49%macro ms_webout(action,ds,dslabel=,fref=_webout,fmt=Y,missing=NULL
50 ,showmeta=NO
51);
52%global _webin_file_count _webin_fileref1 _webin_name1 _program _debug
53 sasjs_tables;
54
55%local i tempds;
56%let action=%upcase(&action);
57
58%if &action=FETCH %then %do;
59 %if %str(&_debug) ge 131 %then %do;
60 options mprint notes mprintnest;
61 %end;
62 %let _webin_file_count=%eval(&_webin_file_count+0);
63 /* now read in the data */
64 %do i=1 %to &_webin_file_count;
65 %if &_webin_file_count=1 %then %do;
66 %let _webin_fileref1=&_webin_fileref;
67 %let _webin_name1=&_webin_name;
68 %end;
69 data _null_;
70 infile &&_webin_fileref&i termstr=crlf lrecl=32767;
71 input;
72 call symputx('input_statement',_infile_);
73 putlog "&&_webin_name&i input statement: " _infile_;
74 stop;
75 data &&_webin_name&i;
76 infile &&_webin_fileref&i firstobs=2 dsd termstr=crlf encoding='utf-8'
77 lrecl=32767;
78 input &input_statement;
79 %if %str(&_debug) ge 131 %then %do;
80 if _n_<20 then putlog _infile_;
81 %end;
82 run;
83 %let sasjs_tables=&sasjs_tables &&_webin_name&i;
84 %end;
85%end;
86
87%else %if &action=OPEN %then %do;
88 /* fix encoding and ensure enough lrecl */
89 OPTIONS NOBOMFILE lrecl=32767;
90
91 /* set the header */
92 %mfs_httpheader(Content-type,application/json)
93
94 /* setup json. */
95 data _null_;file &fref encoding='utf-8' termstr=lf ;
96 put '{"SYSDATE" : "' "&SYSDATE" '"';
97 put ',"SYSTIME" : "' "&SYSTIME" '"';
98 run;
99
100%end;
101
102%else %if &action=ARR or &action=OBJ %then %do;
103 %if "%substr(&sysver,1,1)"="4" or "%substr(&sysver,1,1)"="5" %then %do;
104 /* functions in formats unsupported */
105 %put &sysmacroname: forcing missing back to NULL as feature not supported;
106 %let missing=NULL;
107 %end;
108 %mp_jsonout(&action,&ds,dslabel=&dslabel,fmt=&fmt,jref=&fref
109 ,engine=DATASTEP,missing=&missing,showmeta=&showmeta
110 )
111%end;
112%else %if &action=CLOSE %then %do;
113 %if %str(&_debug) ge 131 %then %do;
114 /* if debug mode, send back first 10 records of each work table also */
115 options obs=10;
116 data;run;%let tempds=%scan(&syslast,2,.);
117 ods output Members=&tempds;
118 proc datasets library=WORK memtype=data;
119 %local wtcnt;%let wtcnt=0;
120 data _null_;
121 set &tempds;
122 if not (upcase(name) =:"DATA"); /* ignore temp datasets */
123 if not (upcase(name)=:"_DATA_");
124 i+1;
125 call symputx(cats('wt',i),name,'l');
126 call symputx('wtcnt',i,'l');
127 data _null_; file &fref mod encoding='utf-8' termstr=lf;
128 put ",""WORK"":{";
129 %do i=1 %to &wtcnt;
130 %let wt=&&wt&i;
131 data _null_; file &fref mod encoding='utf-8' termstr=lf;
132 dsid=open("WORK.&wt",'is');
133 nlobs=attrn(dsid,'NLOBS');
134 nvars=attrn(dsid,'NVARS');
135 rc=close(dsid);
136 if &i>1 then put ','@;
137 put " ""&wt"" : {";
138 put '"nlobs":' nlobs;
139 put ',"nvars":' nvars;
140 %mp_jsonout(OBJ,&wt,jref=&fref,dslabel=first10rows,showmeta=YES)
141 data _null_; file &fref mod encoding='utf-8' termstr=lf;
142 put "}";
143 %end;
144 data _null_; file &fref mod encoding='utf-8' termstr=lf;
145 put "}";
146 run;
147 %end;
148 /* close off json */
149 data _null_;file &fref mod encoding='utf-8' termstr=lf lrecl=32767;
150 _PROGRAM=quote(trim(resolve(symget('_PROGRAM'))));
151 put ",""SYSUSERID"" : ""&sysuserid"" ";
152 put ",""MF_GETUSER"" : ""%mf_getuser()"" ";
153 put ",""_DEBUG"" : ""&_debug"" ";
154 put ',"_PROGRAM" : ' _PROGRAM ;
155 put ",""SYSCC"" : ""&syscc"" ";
156 syserrortext=quote(cats(symget('SYSERRORTEXT')));
157 put ',"SYSERRORTEXT" : ' syserrortext;
158 SYSHOSTINFOLONG=quote(trim(symget('SYSHOSTINFOLONG')));
159 put ',"SYSHOSTINFOLONG" : ' SYSHOSTINFOLONG;
160 put ",""SYSHOSTNAME"" : ""&syshostname"" ";
161 put ",""SYSPROCESSID"" : ""&SYSPROCESSID"" ";
162 put ",""SYSPROCESSMODE"" : ""&SYSPROCESSMODE"" ";
163 length SYSPROCESSNAME $512;
164 SYSPROCESSNAME=quote(urlencode(cats(SYSPROCESSNAME)));
165 put ",""SYSPROCESSNAME"" : " SYSPROCESSNAME;
166 put ",""SYSJOBID"" : ""&sysjobid"" ";
167 put ",""SYSSCPL"" : ""&sysscpl"" ";
168 put ",""SYSSITE"" : ""&syssite"" ";
169 put ",""SYSTCPIPHOSTNAME"" : ""&SYSTCPIPHOSTNAME"" ";
170 sysvlong=quote(trim(symget('sysvlong')));
171 put ',"SYSVLONG" : ' sysvlong;
172 syswarningtext=quote(cats(symget('SYSWARNINGTEXT')));
173 put ',"SYSWARNINGTEXT" : ' syswarningtext;
174 put ',"END_DTTM" : "' "%sysfunc(datetime(),E8601DT26.6)" '" ';
175 length autoexec $512;
176 autoexec=quote(urlencode(trim(getoption('autoexec'))));
177 put ',"AUTOEXEC" : ' autoexec;
178 length memsize $32;
179 memsize="%sysfunc(INPUTN(%sysfunc(getoption(memsize)), best.),sizekmg.)";
180 memsize=quote(cats(memsize));
181 put ',"MEMSIZE" : ' memsize;
182 put "}" @;
183 run;
184%end;
185
186%mend ms_webout;
