1 /**
2  @file
3  @brief Enables previous observations to be re-instated
4  @details Remembers the last X observations by storing them in a hash table.
5  Is a convenience over the use of lag() or retain, when an entire observation
6  needs to be restored.
7 
8  This macro will also restore automatic variables (such as _n_ and _error_).
9 
10  Example Usage:
11 
12 ~~~{.sas}
13 data example;
14  set sashelp.class;
15  calc_var=_n_*3;
16  * initialise hash and save from PDV ;
17  %mp_prevobs(INIT,history=2)
18  if _n_ =10 then do;
19  * fetch previous but 1 record;
20  %mp_prevobs(FETCH,-2)
21  put _n_= name= age= calc_var=;
22  * fetch previous record;
23  %mp_prevobs(FETCH,-1)
24  put _n_= name= age= calc_var=;
25  * reinstate current record ;
26  %mp_prevobs(FETCH,0)
27  put _n_= name= age= calc_var=;
28  end;
29 run;
30 ~~~

51 %macro mp_prevobs(action,record,history=5,prefix=mp_prevobs
52 )/*/STORE SOURCE*/;
53 %let action=%upcase(&action);
54 %let prefix=%upcase(&prefix);
55 %let record=%eval((&record+0) * -1);
56 
57 %if &action=INIT %then %do;
58 
59  if _n_ eq 1 then do;
60  attrib &prefix._VAR length=$64;
61  dcl hash &prefix._HASH(ordered:'Y');
62  &prefix._KEY=0;
63  &prefix._HASH.defineKey("&prefix._KEY");
64  do while(1);
65  call vnext(&prefix._VAR);
66  if &prefix._VAR='' then leave;
67  if &prefix._VAR eq "&prefix._VAR" then continue;
68  else if &prefix._VAR eq "&prefix._KEY" then continue;
69  &prefix._HASH.defineData(&prefix._VAR);
70  end;
71  &prefix._HASH.defineDone();
72  end;
73  /* this part has to happen before FETCHing */
74  &prefix._KEY+1;
75  &prefix._rc=&prefix._HASH.add();
76  if &prefix._rc then putlog 'adding' &prefix._rc=;
77  %if &history>0 %then %do;
78  if &prefix._key>&history+1 then
79  &prefix._HASH.remove(key: &prefix._KEY - &history - 1);
80  if &prefix._rc then putlog 'removing' &prefix._rc=;
81  %end;
82 %end;
83 %else %if &action=FETCH %then %do;
84  if &record > &prefix._key then putlog "Not enough records in &Prefix._hash yet";
85  else &prefix._rc=&prefix._HASH.find(key: &prefix._KEY - &record);
86  if &prefix._rc then putlog &prefix._rc= " when fetching " &prefix._KEY=
87  "with record &record and " _n_=;
88 %end;
89 
90 %mend;
