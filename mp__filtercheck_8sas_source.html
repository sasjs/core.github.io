1/**
2 @file
3 @brief Checks an input filter table for validity
4 @details Performs checks on the input table to ensure it arrives in the
5 correct format. This is necessary to prevent code injection. Will update
6 SYSCC to 1008 if bad records are found, and call mp_abort.sas for a
7 graceful service exit (configurable).
8
9 Used for dynamic filtering in [Data Controller for SAS&reg;](https://datacontroller.io).
10
11 Usage:
12
13 %mp_filtercheck(work.filter,targetds=sashelp.class,outds=work.badrecords)
14
15 The input table should have the following format:
16
17 |GROUP_LOGIC:$3|SUBGROUP_LOGIC:$3|SUBGROUP_ID:8.|VARIABLE_NM:$32|OPERATOR_NM:$10|RAW_VALUE:$4000|
18 |---|---|---|---|---|---|
19 |AND|AND|1|AGE|=|12|
20 |AND|AND|1|SEX|<=|'M'|
21 |AND|OR|2|Name|NOT IN|('Jane','Alfred')|
22 |AND|OR|2|Weight|>=|7|
23
24 Rules applied:
25
26 @li GROUP_LOGIC - only AND/OR
27 @li SUBGROUP_LOGIC - only AND/OR
28 @li SUBGROUP_ID - only integers
29 @li VARIABLE_NM - must be in the target table
30 @li OPERATOR_NM - only =/>/</<=/>=/BETWEEN/IN/NOT IN/NE/CONTAINS
31 @li RAW_VALUE - no unquoted values except integers, commas and spaces.
32
33 @returns The &outds table containing any bad rows, plus a REASON_CD column.
34
35 @param [in] inds The table to be checked, with the format above
36 @param [in] targetds= The target dataset against which to verify VARIABLE_NM.
37 This must be available (ie, the library must be assigned).
38 @param [out] abort= (YES) If YES will call mp_abort.sas on any exceptions
39 @param [out] outds= The output table, which is a copy of the &inds. table
40 plus a REASON_CD column, containing only bad records. If bad records found,
41 the SYSCC value will be set to 1008 (general data problem). Downstream
42 processes should check this table (and return code) before continuing.
43
44 <h4> SAS Macros </h4>
45 @li mp_abort.sas
46 @li mf_getuniquefileref.sas
47 @li mf_getvarlist.sas
48 @li mf_getvartype.sas
49 @li mp_filtergenerate.sas
50 @li mp_filtervalidate.sas
51
52 <h4> Related Macros </h4>
53 @li mp_filtergenerate.sas

55
56 @version 9.3
57 @author Allan Bowe
58
59 @todo Support date / hex / name literals and exponents in RAW_VALUE field
60**/

62%macro mp_filtercheck(inds,targetds=,outds=work.badrecords,abort=YES);
63
64%mp_abort(iftrue= (&syscc ne 0)
65 ,mac=&sysmacroname
66 ,msg=%str(syscc=&syscc - on macro entry)
67)
68
69/* Validate input column */
70%local vtype;
71%let vtype=%mf_getvartype(&inds,RAW_VALUE);
72%mp_abort(iftrue=(&abort=YES and &vtype ne C),
73 mac=&sysmacroname,
74 msg=%str(%str(ERR)OR: RAW_VALUE must be character)
75)
76%if &vtype ne C %then %do;
77 %put &sysmacroname: RAW_VALUE must be character;
78 %let syscc=42;
79 %return;
80%end;
81
82
83/**
84 * Sanitise the values based on valid value lists, then strip out
85 * quotes, commas, periods and spaces.
86 * Only numeric values should remain
87 */
88%local reason_cd nobs;
89%let nobs=0;
90data &outds;
91 /*length GROUP_LOGIC SUBGROUP_LOGIC $3 SUBGROUP_ID 8 VARIABLE_NM $32
92 OPERATOR_NM $10 RAW_VALUE $4000;*/
93 set &inds;
94 length reason_cd $4032;
95
96 /* closed list checks */
97 if GROUP_LOGIC not in ('AND','OR') then do;
98 REASON_CD='GROUP_LOGIC should be AND/OR, not:'!!cats(GROUP_LOGIC);
99 putlog REASON_CD= GROUP_LOGIC=;
100 call symputx('reason_cd',reason_cd,'l');
101 call symputx('nobs',_n_,'l');
102 output;
103 end;
104 if SUBGROUP_LOGIC not in ('AND','OR') then do;
105 REASON_CD='SUBGROUP_LOGIC should be AND/OR, not:'!!cats(SUBGROUP_LOGIC);
106 putlog REASON_CD= SUBGROUP_LOGIC=;
107 call symputx('reason_cd',reason_cd,'l');
108 call symputx('nobs',_n_,'l');
109 output;
110 end;
111 if mod(SUBGROUP_ID,1) ne 0 then do;
112 REASON_CD='SUBGROUP_ID should be integer, not '!!cats(subgroup_id);
113 putlog REASON_CD= SUBGROUP_ID=;
114 call symputx('reason_cd',reason_cd,'l');
115 call symputx('nobs',_n_,'l');
116 output;
117 end;
118 if upcase(VARIABLE_NM) not in
119 (%upcase(%mf_getvarlist(&targetds,dlm=%str(,),quote=SINGLE)))
120 then do;
121 REASON_CD="Variable "!!cats(variable_nm)!!" not in &targetds";
122 putlog REASON_CD= VARIABLE_NM=;
123 call symputx('reason_cd',reason_cd,'l');
124 call symputx('nobs',_n_,'l');
125 output;
126 end;
127 if OPERATOR_NM not in
128 ('=','>','<','<=','>=','BETWEEN','IN','NOT IN','NE','CONTAINS')
129 then do;
130 REASON_CD='Invalid OPERATOR_NM: '!!cats(OPERATOR_NM);
131 putlog REASON_CD= OPERATOR_NM=;
132 call symputx('reason_cd',reason_cd,'l');
133 call symputx('nobs',_n_,'l');
134 output;
135 end;
136
137 /* special logic */
138 if OPERATOR_NM='BETWEEN' then raw_value1=tranwrd(raw_value,' AND ','');
139 else if OPERATOR_NM in ('IN','NOT IN') then do;
140 if substr(raw_value,1,1) ne '('
141 or substr(cats(reverse(raw_value)),1,1) ne ')'
142 then do;
143 REASON_CD='Missing start/end bracket in RAW_VALUE';
144 putlog REASON_CD= OPERATOR_NM= raw_value= raw_value1= ;
145 call symputx('reason_cd',reason_cd,'l');
146 call symputx('nobs',_n_,'l');
147 output;
148 end;
149 else raw_value1=substr(raw_value,2,max(length(raw_value)-2,0));
150 end;
151 else raw_value1=raw_value;
152
153 /* remove nested literals eg '' */
154 raw_value1=tranwrd(raw_value1,"''",'');
155
156 /* now match string literals (always single quotes) */
157 raw_value2=raw_value1;
158 regex = prxparse("s/(\').*?(\')//");
159 call prxchange(regex,-1,raw_value2);
160
161 /* remove commas and periods*/
162 raw_value3=compress(raw_value2,',.');
163
164 /* output records that contain values other than digits and spaces */
165 if notdigit(compress(raw_value3,' '))>0 then do;
166 putlog raw_value3= $hex32.;
167 REASON_CD=cats('Invalid RAW_VALUE:',raw_value);
168 putlog REASON_CD= raw_value= raw_value1= raw_value2= raw_value3=;
169 call symputx('reason_cd',reason_cd,'l');
170 call symputx('nobs',_n_,'l');
171 output;
172 end;
173
174run;
175
176
177data _null_;
178 set &outds end=last;
179 putlog (_all_)(=);
180run;
181
182%mp_abort(iftrue=(&abort=YES and &nobs>0),
183 mac=&sysmacroname,
184 msg=%str(Data issue: %superq(reason_cd))
185)
186
187%if &nobs>0 %then %do;
188 %let syscc=1008;
189 %return;
190%end;
191
192/**
193 * syntax checking passed but it does not mean the filter is valid
194 * for that we can run a proc sql validate query
195 */
196%local fref1;
197%let fref1=%mf_getuniquefileref();
198%mp_filtergenerate(&inds,outref=&fref1)
199
200/* this macro will also set syscc to 1008 if any issues found */
201%mp_filtervalidate(&fref1,&targetds,outds=&outds,abort=&abort)
202
203%mend mp_filtercheck;
