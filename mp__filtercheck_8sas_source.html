1 /**
2  @file
3  @brief Checks an input filter table for validity
4  @details Performs checks on the input table to ensure it arrives in the
5  correct format. This is necessary to prevent code injection. Will update
6  SYSCC to 1008 if bad records are found.
7 
8  Used for dynamic filtering in [Data Controller for SAS&reg;](https://datacontroller.io).
9 
10  Usage:
11 
12  %mp_filtercheck(work.filter,targetds=sashelp.class,outds=work.badrecords)
13 
14  The input table should have the following format:
15 
16  |GROUP_LOGIC:$3|SUBGROUP_LOGIC:$3|SUBGROUP_ID:8.|VARIABLE_NM:$32|OPERATOR_NM:$10|RAW_VALUE:$32767|
17  |---|---|---|---|---|---|
18  |AND|AND|1|AGE|=|12|
19  |AND|AND|1|SEX|<=|'M'|
20  |AND|OR|2|Name|NOT IN|('Jane','Alfred')|
21  |AND|OR|2|Weight|>=|7|
22 
23  Rules applied:
24 
25  @li GROUP_LOGIC - only AND/OR
26  @li SUBGROUP_LOGIC - only AND/OR
27  @li SUBGROUP_ID - only integers
28  @li VARIABLE_NM - must be in the target table
29  @li OPERATOR_NM - only =/>/</<=/>=/BETWEEN/IN/NOT IN/NOT EQUAL/CONTAINS
30  @li RAW_VALUE - no unquoted values except integers, commas and spaces.
31 
32  @returns The &outds table containing any bad rows, plus a REASON_CD column.
33 
34  @param [in] inds The table to be checked, with the format above
35  @param [in] targetds= The target dataset against which to verify VARIABLE_NM
36  @param [out] outds= The output table, which is a copy of the &inds. table
37  plus a REASON_CD column, containing only bad records. If bad records found,
38  the SYSCC value will be set to 1008 (general data problem). Downstream
39  processes should check this table (and return code) before continuing.
40 
41  <h4> SAS Macros </h4>
42  @li mp_abort.sas
43  @li mf_getvarlist.sas
44  @li mf_nobs.sas
45 
46  <h4> Related Macros </h4>
47  @li mp_filtergenerate.sas
48 
49  @version 9.3
50  @author Allan Bowe
51 
52  @todo Support date / hex / name literals and exponents in RAW_VALUE field
53 **/

55 %macro mp_filtercheck(inds,targetds=,outds=work.badrecords);
56 
57 %mp_abort(iftrue= (&syscc ne 0)
58  ,mac=&sysmacroname
59  ,msg=%str(syscc=&syscc - on macro entry)
60 )
61 
62 /**
63  * Sanitise the values based on valid value lists, then strip out
64  * quotes, commas, periods and spaces.
65  * Only numeric values should remain
66  */
67 
68 data &outds;
69  set &inds;
70  length reason_cd $32;
71 
72  /* closed list checks */
73  if GROUP_LOGIC not in ('AND','OR') then do;
74  REASON_CD='GROUP_LOGIC should be either AND or OR';
75  putlog REASON_CD= GROUP_LOGIC=;
76  output;
77  end;
78  if SUBGROUP_LOGIC not in ('AND','OR') then do;
79  REASON_CD='SUBGROUP_LOGIC should be either AND or OR';
80  putlog REASON_CD= SUBGROUP_LOGIC=;
81  output;
82  end;
83  if mod(SUBGROUP_ID,1) ne 0 then do;
84  REASON_CD='SUBGROUP_ID should be integer';
85  putlog REASON_CD= SUBGROUP_ID=;
86  output;
87  end;
88  if upcase(VARIABLE_NM) not in
89  (%upcase(%mf_getvarlist(&targetds,dlm=%str(,),quote=SINGLE)))
90  then do;
91  REASON_CD="VARIABLE_NM not in &targetds";
92  putlog REASON_CD= VARIABLE_NM=;
93  output;
94  end;
95  if OPERATOR_NM not in
96  ('=','>','<','<=','>=','BETWEEN','IN','NOT IN','NOT EQUAL','CONTAINS')
97  then do;
98  REASON_CD='Invalid OPERATOR_NM';
99  putlog REASON_CD= OPERATOR_NM=;
100  output;
101  end;
102 
103  /* special logic */
104  if OPERATOR_NM='BETWEEN' then raw_value1=tranwrd(raw_value,' AND ','');
105  else if OPERATOR_NM in ('IN','NOT IN') then do;
106  if substr(raw_value,1,1) ne '('
107  or substr(cats(reverse(raw_value)),1,1) ne ')'
108  then do;
109  REASON_CD='Missing brackets in RAW_VALUE';
110  putlog REASON_CD= OPERATOR_NM= raw_value= raw_value1= ;
111  output;
112  end;
113  else raw_value1=substr(raw_value,2,max(length(raw_value)-2,0));
114  end;
115  else raw_value1=raw_value;
116 
117  /* remove nested literals eg '' */
118  raw_value1=tranwrd(raw_value1,"''",'');
119 
120  /* now match string literals (always single quotes) */
121  raw_value2=raw_value1;
122  regex = prxparse("s/(\').*?(\')//");
123  call prxchange(regex,-1,raw_value2);
124 
125  /* remove commas */
126  raw_value3=compress(raw_value2,',');
127 
128 
129 
130 
131  /* output records that contain values other than digits and spaces */
132  if notdigit(compress(raw_value3,' '))>0 then do;
133  putlog raw_value3= $hex32.;
134  REASON_CD='Invalid RAW_VALUE';
135  putlog REASON_CD= raw_value= raw_value1= raw_value2= raw_value3=;
136  output;
137  end;
138 
139 run;
140 
141 %if %mf_nobs(&outds)>0 %then %let syscc=1008;
142 
143 %mend;
