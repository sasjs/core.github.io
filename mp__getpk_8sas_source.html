1/**
2 @file
3 @brief Extract the primary key fields from a table or library
4 @details Examines the constraints to identify primary key fields - indicated
5 by an explicit PK constraint, or a unique index that is also NOT NULL.
6
7 Can be executed at both table and library level. Supports both BASE engine
8 libraries and SQL Server.
9
10 Usage:
11
12 proc sql;
13 create table work.example(
14 TX_FROM float format=datetime19.,
15 DD_TYPE char(16),
16 DD_SOURCE char(2048),
17 DD_SHORTDESC char(256),
18 constraint pk primary key(tx_from, dd_type,dd_source),
19 constraint unq unique(tx_from, dd_type),
20 constraint nnn not null(DD_SHORTDESC)
21 );
22 %mp_getpk(work,ds=example)
23
24 Returns:
25
26|libref:$8.|dsn:$32.|memtype:$8.|dbms_memtype:$32.|typemem:$8.|memlabel:$256.|nvar:best.|compress:$8.|pk_fields:$512.|
27|---|---|---|---|---|---|---|---|---|
28|WORK|EXAMPLE|DATA||DATA||4|NO|TX_FROM DD_TYPE DD_SOURCE|
29
30
31 @param [in] lib The libref to examine
32 @param [in] ds= (0) Select the dataset to examine, else use 0 for all tables
33 @param [in] mdebug= (0) Set to 1 to preserve temp tables, print var values etc
34 @param [out] outds= (work.mp_getpk) The name of the output table to create.
35
36 <h4> SAS Macros </h4>
37 @li mf_getengine.sas
38 @li mf_getschema.sas
39 @li mp_dropmembers.sas
40 @li mp_getconstraints.sas
41
42 <h4> Related Macros </h4>

50%macro mp_getpk(

52 ds=0,
53 outds=work.mp_getpk,

58%local engine schema ds1 ds2 ds3 dsn tabs1 tabs2 sum pk4sure pkdefault finalpks;
59
60%let lib=%upcase(&lib);
61%let ds=%upcase(&ds);
62%let engine=%mf_getengine(&lib);
63%let schema=%mf_getschema(&lib);
64
65%let ds1=%mf_getuniquename(prefix=getpk_ds1);
66%let ds2=%mf_getuniquename(prefix=getpk_ds2);
67%let ds3=%mf_getuniquename(prefix=getpk_ds3);
68%let tabs1=%mf_getuniquename(prefix=getpk_tabs1);
69%let tabs2=%mf_getuniquename(prefix=getpk_tabs2);
70%let sum=%mf_getuniquename(prefix=getpk_sum);
71%let pk4sure=%mf_getuniquename(prefix=getpk_pk4sure);
72%let pkdefault=%mf_getuniquename(prefix=getpk_pkdefault);
73%let finalpks=%mf_getuniquename(prefix=getpk_finalpks);

75%local dbg;
76%if &mdebug=1 %then %do;
77 %put &sysmacroname entry vars:;
78 %put _local_;
79%end;

82proc sql;
83create table &ds1 as
84 select libname as libref
85 ,upcase(memname) as dsn
86 ,memtype
87 ,upcase(name) as name
88 ,type
89 ,length
90 ,varnum
91 ,label
92 ,format
93 ,idxusage
94 ,notnull
95 from dictionary.columns
96 where upcase(libname)="&lib"
97%if &ds ne 0 %then %do;
98 and upcase(memname)="&ds"
99%end;
100 ;

103%if &engine=SQLSVR %then %do;
104 proc sql;
105 connect using &lib;
106 create table work.&ds2 as
107 select * from connection to &lib(
108 select
109 s.name as SchemaName,
110 t.name as memname,
111 tc.name as name,
112 ic.key_ordinal as KeyOrderNr
113 from
114 sys.schemas s
115 inner join sys.tables t on s.schema_id=t.schema_id
116 inner join sys.indexes i on t.object_id=i.object_id
117 inner join sys.index_columns ic on i.object_id=ic.object_id
118 and i.index_id=ic.index_id
119 inner join sys.columns tc on ic.object_id=tc.object_id
120 and ic.column_id=tc.column_id
121 where i.is_primary_key=1
122 and s.name=%str(%')&schema%str(%')
123 order by t.name, ic.key_ordinal ;
124 );disconnect from &lib;
125 create table &ds3 as
126 select a.*
127 ,case when b.name is not null then 1 else 0 end as pk_ind
128 from work.&ds1 a
129 left join work.&ds2 b
130 on a.dsn=b.memname
131 and upcase(a.name)=upcase(b.name)
132 order by libref,dsn;
133%end;
134%else %do;

136 %if &ds = 0 %then %let dsn=;

138 /* get all constraints, in constraint order*/
139 %mp_getconstraints(lib=&lib,ds=&dsn,outds=work.&ds2)
140
141 /* extract cols that are clearly primary keys */
142 proc sql;
143 create table &pk4sure as
144 select libref
145 ,table_name
146 ,constraint_name
147 ,constraint_order
148 ,column_name as name
149 from work.&ds2
150 where constraint_type='PRIMARY'
151 order by 1,2,3,4;
152
153 /* extract unique constraints where every col is also NOT NULL */
154 proc sql;
155 create table &sum as
156 select a.libref
157 ,a.table_name
158 ,a.constraint_name
159 ,count(a.column_name) as unq_cnt
160 ,count(b.column_name) as nul_cnt
161 from work.&ds2(where=(constraint_type ='UNIQUE')) a
162 left join work.&ds2(where=(constraint_type ='NOT NULL')) b
163 on a.libref=b.libref
164 and a.table_name=b.table_name
165 and a.column_name=b.column_name
166 group by 1,2,3
167 having unq_cnt=nul_cnt;
168
169 /* extract cols from the relevant unique constraints */
170 create table &pkdefault as
171 select a.libref
172 ,a.table_name
173 ,a.constraint_name
174 ,b.constraint_order
175 ,b.column_name as name
176 from &sum a
177 left join &ds2(where=(constraint_type ='UNIQUE')) b
178 on a.libref=b.libref
179 and a.table_name=b.table_name
180 and a.constraint_name=b.constraint_name
181 order by 1,2,3,4;
182
183 /* create one table */
184 data &finalpks;
185 set &pkdefault &pk4sure ;
186 pk_ind=1;
187 /* if there are multiple unique constraints, take the first */
188 by libref table_name constraint_name;
189 retain keepme;
190 if first.table_name then keepme=1;
191 if first.constraint_name and not first.table_name then keepme=0;
192 if keepme=1;
193 run;
194
195 /* join back to starting table */
196 proc sql;
197 create table &ds3 as
198 select a.*
199 ,b.constraint_order
200 ,case when b.pk_ind=1 then 1 else 0 end as pk_ind
201 from work.&ds1 a
202 left join work.&finalpks b
203 on a.libref=b.libref
204 and a.dsn=b.table_name
205 and upcase(a.name)=upcase(b.name)
206 order by libref,dsn,constraint_order;
207%end;

210/* prepare tables */
211proc sql;
212create table work.&tabs1 as select
213 libname as libref
214 ,upcase(memname) as dsn
215 ,memtype
216 ,dbms_memtype
217 ,typemem
218 ,memlabel
219 ,nvar
220 ,compress
221from dictionary.tables
222 where upcase(libname)="&lib"
223%if &ds ne 0 %then %do;
224 and upcase(memname)="&ds"
225%end;
226 ;
227data &tabs2;
228 set &ds3;
229 length pk_fields $512;
230 retain pk_fields;
231 by libref dsn constraint_order;
232 if first.dsn then pk_fields='';
233 if pk_ind=1 then pk_fields=catx(' ',pk_fields,name);
234 if last.dsn then output;
235run;
236
237proc sql;
238create table &outds as
239 select a.libref
240 ,a.dsn
241 ,a.memtype
242 ,a.dbms_memtype
243 ,a.typemem
244 ,a.memlabel
245 ,a.nvar
246 ,a.compress
247 ,b.pk_fields
248 from work.&tabs1 a
249 left join work.&tabs2 b
250 on a.libref=b.libref
251 and a.dsn=b.dsn;

253/* tidy up */
254%mp_dropmembers(

257)
258
259%mend mp_getpk;
